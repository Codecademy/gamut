For Grid layouts we&apos;ve added a specific flex <code>GridBox</code>
component that extends <LinkTo component="Box">Box</LinkTo> with added flex props.

<Canvas>
  <Story
    name="GridBox"
    args={{
      rowGap: 16,
    }}
  >
    {(args) => {
      const elProps = {
        border: 1,
        p: 16,
      };
      return (
        <GridBox {...args}>
          <Box {...elProps}>Child Element 1</Box>
          <Box {...elProps}>Child Element 2</Box>
        </GridBox>
      );
    }}
  </Story>
</Canvas>
