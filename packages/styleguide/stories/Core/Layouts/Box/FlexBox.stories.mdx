# FlexBox

<StoryStatus status="volatile" />

<p>
  For flex layouts we've added a specific flex <code>FlexBox</code> component
  that extends{' '}
  <code>
    <LinkTo kind="Layouts/Box">Box</LinkTo>
  </code>{' '}
  with added flex props.
</p>

<Canvas>
  <Story
    name="FlexBox"
    args={{
      children: 'Hello world I am a FlexBox',
    }}
  >
    {({ as = 'div', ...args }) => {
      const elProps = {
        borderWidth: '1px',
        borderStyle: 'solid',
        borderColor: 'navy',
        padding: 16,
      };
      return (
        <FlexBox {...args}>
          <Box {...elProps}>Child Element 1</Box>
          <Box {...elProps}>Child Element 2</Box>
        </FlexBox>
      );
    }}
  </Story>
</Canvas>
