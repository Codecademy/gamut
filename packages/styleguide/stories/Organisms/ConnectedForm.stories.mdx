## Usage

```tsx
const VeryCoolForm = () => {
  const {
    ConnectedFormGroup,
    ConnectedForm,
    connectedFormProps,
  } = useConnectedForm({
    defaultValues: {
      thisField: true,
      thatField: 'zero!',
      anotherField: 'state your name',
    },
    validationRules: {
      thisField: { required: 'yes' },
      thatField: {
        pattern: {
          value: /^(?:(?!zero).)*$/,
          message: 'notZero',
        },
      },
    },
  });

  return (
    <Background bg="black">
      <ConnectedForm
        onSubmit={({ thisField }) => console.log(thisField)}
        {...connectedFormProps}
      >
        <ConnectedFormGroup
          name="thisField"
          label="cool"
          field={{
            component: ConnectedCheckbox,
            label: 'check it ouuuut',
          }}
        />
        <Box>
          <Text>I have something important to tell you.</Text>
        </Box>
        <ConnectedFormGroup
          name="thatField"
          label="cool"
          field={{
            component: ConnectedSelect,
            options: ['one', 'two', 'zero'],
          }}
        />
        <ConnectedFormGroup
          name="anotherField"
          label="cool"
          field={{
            component: ConnectedInput,
            icon: TerminalIcon,
          }}
        />
      </ConnectedForm>
    </Background>
  );
};
};
```
