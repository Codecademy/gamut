<Meta

  component={FocusTrap}

    source: 'gamut',
    status: 'current',
  }}

<Canvas>
  <Story name="FocusTrap">

              >
                <GridForm
                  fields={[
                    {

                      size: 12,
                      type: 'text',
                    },
                    {
                      label: 'password',
                      name: 'password-text',
                      size: 12,
                      type: 'password',
                    },
                  ]}
                  onSubmit={async () => {
                    setActive(false);
                  }}
                  submit={{
                    contents: 'Disable Focus Trap',
                    position: 'right',
                    size: 12,
                  }}
                />
              </Box>
            </FocusTrap>

        </>

    }}
  </Story>
