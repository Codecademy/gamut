### iFrame Inputs

sometime you gotta use iframes. it's okay, we understand. and we have styles for it! in order to use them, you need to do something like this:

```tsx
import { Input, ReactRecurlyStyles } from '@codecademy/gamut';
import styled from '@emotion/styled';
import { GnarlyiFrame } from './constants';

const StylediFrame = styled(GnarlyiFrame)`
  ${ReactRecurlyStyles}
`;

export const SomeWrapper: React.FC = ({ props }) => {
  return (
    <>
      <Input
        component={StylediFrame}
        onChange={() => {}}
        onBlur={() => {}}
        placeholder="Placeholder"
      />
    </>
  );
};
```
